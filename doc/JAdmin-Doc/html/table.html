表结构文档

JAdmin需要11张基础表支撑

sys_config 系统配置表
字段名称 字段类型 相关属性 字段说明
configId varchar(32) PRIMARY KEY 主键
name varchar(128) NOT NULL 名字
code varchar(128) NOT NULL 编号
coValue text NOT NULL 值
isOpen varchar(128) NOT NULL 是否允许在界面中编辑 1-是，0-否
operateTime varchar(19) NOT NULL 操作时间
operatorId varchar(32) NOT NULL 操作人
billStatus char(1) NOT NULL 状态 1正常 0禁用
memo varchar(512) NULL 描述
sys_dictkind 字典种类表
字段名称 字段类型 相关属性 字段说明
dictkindId varchar(32) PRIMARY KEY 主键
code varchar(128) NOT NULL 字典种类编号
name varchar(1024) NOT NULL 字典种类名称
isDynamic char(1) NOT NULL 是否是动态的，动态查询数据库的为1，否则0
dynSql varchar(1024) NULL 如果字典的属性值不是固定的，就查询sql
type char(1) NOT NULL 字典类别类型，多选还是单选，详见数据字典
isOpen char(1) NOT NULL 是否允许在界面中编辑 1-是，0-否
operateTime varchar(19) NULL 操作时间
billStatus char(1) NOT NULL 状态 1正常 0禁用
memo varchar(512) NULL 描述
sys_dictinfo 字典属性表
字段名称 字段类型 相关属性 字段说明
dictinfoId varchar(32) PRIMARY KEY 主键
dictkindId varchar(32) NULL 字典父ID
dictinfoFid varchar(32) NULL 如果是2级以上的，需要fid
name varchar(1024) NOT NULL 字典名字
code varchar(128) NULL 字典编号
sort int(10) NOT NULL 排序，小的在前面
level char(1) NOT NULL 级别，默认1级
billStatus char(1) NOT NULL 状态 1正常 0禁用
memo varchar(512) NULL 描述
sys_log 系统日志表
字段名称 字段类型 相关属性 字段说明
logId varchar(32) PRIMARY KEY 主键
orgId varchar(32) NULL 机构ID
userId varchar(32) NULL 用户ID
url varchar(1024) NULL 请求的url
type varchar(128) NULL 类型
content longtext NOT NULL 日志内容
clientIp varchar(64) NULL 用户操作IP
level char(1) NULL 日志级别
operateTime varchar(19) NULL 操作时间
memo varchar(512) NULL 描述
sys_memorandum 备忘笺
字段名称 字段类型 相关属性 字段说明
memorandumId varchar(32) PRIMARY KEY 主键
content text NULL 备忘签内容
isDelete char(1) NOT NULL 是否删除 1-是 0-否
operateTime varchar(19) NOT NULL 操作时间
operatorId varchar(32) NOT NULL 操作人
sys_org 部门表
字段名称 字段类型 相关属性 字段说明
orgId varchar(32) PRIMARY KEY 主键
code varchar(128) NOT NULL 字典种类编号
name varchar(1024) NOT NULL 字典种类名称
fullname varchar(1024) NULL 全称
seq varchar(1024) NOT NULL 序列号
orgFSeq varchar(1024) NOt NULL 所属父seq
isDelete char(1) NOT NULL 是否删除
operateTime varchar(19) NOT NULL 操作时间
operatorId varchar(32) NOT NULL 操作人
billStatus char(1) NOT NULL 状态 1正常 0禁用
memo varchar(512) NULL 描述
sys_user 系统用户表
字段名称 字段类型 相关属性 字段说明
userId varchar(32) PRIMARY KEY 主键
orgId varchar(32) NULL 所属机构/部门
roleId varchar(32) NOT NULL 所属角色
account varchar(128) NOT NULL 账号
name varchar(128) NOT NULL 名字
password varchar(128) NOT NULL 密码
sex char(1) NULL 性别，详见数据词典
lastLoginIp varchar(32) NULL 上次登录ip
lastLoginTime varchar(19) NULL 上次登录时间
loginCount int(10) NOT NULL 登录次数
psChangeTime varchar(19) NULL 密码修改时间
isDelete char(1) NOT NULL 是否删除
operateTime varchar(19) NOT NULL 操作时间
operatorId varchar(32) NOT NULL 操作人
billStatus char(1) NOT NULL 状态 1正常 0禁用
memo varchar(512) NULL 描述
sys_role 角色表
字段名称 字段类型 相关属性 字段说明
roleId varchar(32) PRIMARY KEY 主键
roleName varchar(128) NOT NULL 角色名称
roleType char(1) NOT NULL 1全局型角色、2应用级角色
operateTime varchar(19) NOT NULL 操作时间
operatorId varchar(32) NOT NULL 操作人
billStatus char(1) NOT NULL 状态 1正常 0禁用
memo varchar(512) NULL 描述
sys_role_module 角色权限表
字段名称 字段类型 相关属性 字段说明
roleModuleId varchar(32) PRIMARY KEY 主键
roleId varchar(128) NOT NULL 角色名称
roleType char(1) NOT NULL 1全局型角色、2应用级角色
operateTime varchar(19) NOT NULL 操作时间
operatorId varchar(32) NOT NULL 操作人
billStatus char(1) NOT NULL 状态 1正常 0禁用
memo varchar(512) NULL 描述
sys_task 系统调度任务表
字段名称 字段类型 相关属性 字段说明
taskId varchar(32) PRIMARY KEY 主键
className varchar(256) NOT NULL 包名+类名
cron varchar(256) NULL cron表达式
initialDelay int(32) NULL 延迟执行 单位秒
description varchar(256) NULL 描述
serverIp varchar(32) NULL 服务器ip
billStatus char(1) NULL 调度开关，详见数据词典
isDynamic char(1) NULL 是否可修改 1-是 0-否
isEdited char(1) NULL 是否已经在界面中编辑过
sys_task_log 调度任务执行记录表
字段名称 字段类型 相关属性 字段说明
logId varchar(32) PRIMARY KEY 主键
className varchar(256) NOT NULL 包名+类名
serverIp varchar(32) NOT NULL 服务器ip
content varchar(2048) NOT NULL 内容
startTime varchar(19) NOT NULL 开始时间
endTime varchar(19) NOT NULL 结束时间
executeTime int(11) NOT NULL 耗时 单位为秒
